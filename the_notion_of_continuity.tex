\section{The Notion of Continuity}

\begin{definition}
	Let $X$ and $Y$ be two topological spaces and $f: X \to Y$. The map $f$ is said to be \bld{continuous} if for any open set $U \subseteq Y$ we have that $f^{-1}(Y)$ is open in $X$.
\end{definition}
